import pandas as pd
import re
df = pd.read_csv('/content/reviews_sample (1).csv')


# Positivas
positive_words = [
    'beautiful','amazing','friendly','great','excellent','wonderful','delicious',
    'stunning','fantastic','perfect','incredible','love','clean','enjoyable',
    'relaxing','helpful','unforgettable','fun','charming','recommend'
]

# Negativas
negative_words = [
    'dirty','terrible','bad','rude','expensive','crowded','worst','uncomfortable',
    'disappointed','slow','boring','overpriced','poor','unfriendly','noisy',
    'problem','confusing','small','unhelpful','avoid'
]

# Crear columna limpia del texto (todo minsculas y sin NaN)
df['text_clean'] = df['text'].fillna('').str.lower()
df['positive_count'] = df['text_clean'].apply(lambda x: count_word_list(x, positive_words))
df['negative_count'] = df['text_clean'].apply(lambda x: count_word_list(x, negative_words))

total_positive = df['positive_count'].sum()
total_negative = df['negative_count'].sum()

print(f"Total palabras positivas: {total_positive}")
print(f"Total palabras negativas: {total_negative}")

from collections import Counter
# Asegrate de tener la columna text_clean
df['text_clean'] = df['text'].fillna('').str.lower()

# Funcin para contar palabras de la lista
def count_words_frequency(text_series, word_list):
    counter = Counter()
    for text in text_series:
        words = re.findall(r'\b\w+\b', text)
        counter.update([w for w in words if w in word_list])
    return counter

# Contar frecuencia de palabras positivas y negativas
positive_freq = count_words_frequency(df['text_clean'], positive_words)
negative_freq = count_words_frequency(df['text_clean'], negative_words)

# Convertir a DataFrame para ranking
positive_df = pd.DataFrame(positive_freq.most_common(), columns=['word', 'count'])
negative_df = pd.DataFrame(negative_freq.most_common(), columns=['word', 'count'])
import matplotlib.pyplot as plt

# Top 10 palabras positivas
top_pos = positive_df.head(10)
plt.figure(figsize=(7,7))
plt.pie(top_pos['count'], labels=top_pos['word'], autopct='%1.1f%%', startangle=140)
plt.title('Top 10 palabras positivas en reviews de Lisboa')
plt.show()

# Top 10 palabras negativas
top_neg = negative_df.head(10)
plt.figure(figsize=(7,7))
plt.pie(top_neg['count'], labels=top_neg['word'], autopct='%1.1f%%', startangle=140)
plt.title('Top 10 palabras negativas en reviews de Lisboa')
plt.show()

import re
from collections import Counter

def count_words_lang(text, word_list):
    text = str(text).lower()
    words = re.findall(r'\b\w+\b', text)
    return sum(1 for w in words if w in word_list)

# Creamos columna de conteo por idioma (ejemplo ingls)

            This defaults to the reveal CDN, but can be any url pointing to a copy
            of reveal.js.
            For speaker notes to work, this must be a relative path to a local
            copy of reveal.js: e.g., "reveal.js".
            If a relative path is given, it must be a subdirectory of the
            current directory (from which the server is run).
            See the usage documentation
            (https://nbconvert.readthedocs.io/en/latest/usage.html#reveal-js-html-slideshow)
            for more details.
    Default: ''
    Equivalent to: [--SlidesExporter.reveal_url_prefix]
--nbformat=<Enum>
    The nbformat version to write.
            Use this to downgrade notebooks.
    Choices: any of [1, 2, 3, 4]
    Default: 4
    Equivalent to: [--NotebookExporter.nbformat_version]

Examples
--------

    The simplest way to use nbconvert is

            > jupyter nbconvert mynotebook.ipynb --to html

            Options include ['asciidoc', 'custom', 'html', 'latex', 'markdown', 'notebook', 'pdf', 'python', 'qtpdf', 'qtpng', 'rst', 'script', 'slides', 'webpdf'].

            > jupyter nbconvert --to latex mynotebook.ipynb

            Both HTML and LaTeX support multiple output templates. LaTeX includes
            'base', 'article' and 'report'.  HTML includes 'basic', 'lab' and
            'classic'. You can specify the flavor of the format used.

            > jupyter nbconvert --to html --template lab mynotebook.ipynb

            You can also pipe the output to stdout, rather than a file

            > jupyter nbconvert mynotebook.ipynb --stdout

            PDF is generated via latex

            > jupyter nbconvert mynotebook.ipynb --to pdf

            You can get (and serve) a Reveal.js-powered slideshow

            > jupyter nbconvert myslides.ipynb --to slides --post serve

            Multiple notebooks can be given at the command line in a couple of
            different ways:

            > jupyter nbconvert notebook*.ipynb
            > jupyter nbconvert notebook1.ipynb notebook2.ipynb

            or you can specify the notebooks list in a config file, containing::

                c.NbConvertApp.notebooks = ["my_notebook.ipynb"]

            > jupyter nbconvert --config mycfg.py

To see all available configurables, use `--help-all`.
